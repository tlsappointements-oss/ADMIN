Phone: ${phone}
Destination: ${dest}
Visa Type: ${vtype}
Travel Date: ${tdate}
Applicants: ${num}
Package: ${pkgName || '—'}
Add-ons: ${addons}
Total: ${pkgPrice + addonTotal} MAD `; } // Submit application window.submitApplication = async function() { const { db, collection, addDoc, serverTimestamp } = window._fb; const fn = document.getElementById('fname').value.trim(); const ln = document.getElementById('lname').value.trim(); const email = document.getElementById('app-email').value.trim(); if (!fn || !email) { showNotif('⚠️', 'Please fill in at least name and email'); return; } const ref = 'EVP-' + new Date().getFullYear() + '-' + Math.floor(1000 + Math.random()*9000); const data = { ref, firstName: fn, lastName: ln, email, phone: document.getElementById('phone').value, destination: document.getElementById('dest-country').value, visaType: document.getElementById('visa-type').value, travelDate: document.getElementById('travel-date').value, numApplicants: document.getElementById('num-applicants').value, package: pkgName, addons: Array.from(document.querySelectorAll('.addon-item input:checked')).map(c => c.dataset.name), total: pkgPrice + addonTotal, notes: document.getElementById('notes').value, status: 'pending', createdAt: serverTimestamp(), userId: window._fb.auth.currentUser ? window._fb.auth.currentUser.uid : 'guest', }; try { await addDoc(collection(db, 'applications'), data); showNotif('✅', `Application submitted!
